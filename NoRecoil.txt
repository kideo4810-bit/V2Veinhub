-- StarterPlayer > StarterPlayerScripts > LocalScript
-- Dev test only: Local-only no recoil patcher for your own game

local Players = game:GetService("Players")
local UIS = game:GetService("UserInputService")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local playerGui = player:WaitForChild("PlayerGui")

local enabled = false

local originals = {}
local patchedTools = {}

local MODULE_NAMES = {
	Config = true,
	PassConfig = true,
	Gun = true,
}

local FLAT_PATCH = {
	Recoil = 0,
	AngleX_Min = 0,
	AngleX_Max = 0,
	AngleY_Min = 0,
	AngleY_Max = 0,
	AngleZ_Min = 0,
	AngleZ_Max = 0,
	RecoilSpeed = 0,
	RecoilDamper = 1,
	RecoilRedution = 1,
	Spread = 0,
	CrossExpansion = 0,
}

local AIM_PATCH = {
	RecoilReduction_X = 0,
	RecoilReduction_Y = 0,
	RecoilReduction_Z = 0,
}

local function saveOriginal(tbl, key, value)
	if originals[tbl] == nil then
		originals[tbl] = {}
	end
	if originals[tbl][key] == nil then
		originals[tbl][key] = value
	end
end

local function patchTable(tbl)
	if type(tbl) ~= "table" then
		return false
	end

	local changed = false

	for key, value in pairs(FLAT_PATCH) do
		if tbl[key] ~= nil then
			saveOriginal(tbl, key, tbl[key])
			tbl[key] = value
			changed = true
		end
	end

	if type(tbl.AimSettings) == "table" then
		for key, value in pairs(AIM_PATCH) do
			if tbl.AimSettings[key] ~= nil then
				saveOriginal(tbl.AimSettings, key, tbl.AimSettings[key])
				tbl.AimSettings[key] = value
				changed = true
			end
		end
	end

	return changed
end

local function restoreTable(tbl)
	local saved = originals[tbl]
	if not saved then
		return
	end

	for key, value in pairs(saved) do
		tbl[key] = value
	end

	originals[tbl] = nil
end

local function tryRequire(mod)
	local ok, result = pcall(require, mod)
	if not ok then
		return nil
	end
	return result
end

local function patchTool(tool)
	if not tool or not tool:IsA("Tool") then
		return
	end

	if patchedTools[tool] then
		return
	end

	for _, child in ipairs(tool:GetChildren()) do
		if child:IsA("ModuleScript") and MODULE_NAMES[child.Name] then
			local result = tryRequire(child)
			if type(result) == "table" then
				patchTable(result)
			end
		end
	end

	patchedTools[tool] = true
end

local function restoreTool(tool)
	if not tool or not tool:IsA("Tool") then
		return
	end

	for _, child in ipairs(tool:GetChildren()) do
		if child:IsA("ModuleScript") and MODULE_NAMES[child.Name] then
			local result = tryRequire(child)
			if type(result) == "table" then
				restoreTable(result)
			end
		end
	end

	patchedTools[tool] = nil
end

local function getAllTools()
	local list = {}

	local backpack = player:FindFirstChild("Backpack")
	if backpack then
		for _, obj in ipairs(backpack:GetChildren()) do
			if obj:IsA("Tool") then
				table.insert(list, obj)
			end
		end
	end

	local char = player.Character
	if char then
		for _, obj in ipairs(char:GetChildren()) do
			if obj:IsA("Tool") then
				table.insert(list, obj)
			end
		end
	end

	return list
end

local function applyState()
	for _, tool in ipairs(getAllTools()) do
		if enabled then
			patchTool(tool)
		else
			restoreTool(tool)
		end
	end
end

-- UI
local gui = Instance.new("ScreenGui")
gui.Name = "LocalNoRecoilUI"
gui.ResetOnSpawn = false
gui.Parent = playerGui

local frame = Instance.new("Frame")
frame.Size = UDim2.fromOffset(170, 68)
frame.Position = UDim2.fromOffset(40, 180)
frame.BackgroundColor3 = Color3.fromRGB(24, 24, 24)
frame.BorderSizePixel = 0
frame.Parent = gui

local frameCorner = Instance.new("UICorner")
frameCorner.CornerRadius = UDim.new(0, 8)
frameCorner.Parent = frame

local frameStroke = Instance.new("UIStroke")
frameStroke.Transparency = 0.35
frameStroke.Parent = frame

local title = Instance.new("TextLabel")
title.BackgroundTransparency = 1
title.Size = UDim2.new(1, -10, 0, 22)
title.Position = UDim2.fromOffset(5, 4)
title.Text = "Local No Recoil"
title.TextColor3 = Color3.fromRGB(255, 255, 255)
title.TextSize = 14
title.Font = Enum.Font.GothamSemibold
title.Parent = frame

local button = Instance.new("TextButton")
button.Size = UDim2.new(1, -12, 0, 30)
button.Position = UDim2.fromOffset(6, 30)
button.BackgroundColor3 = Color3.fromRGB(80, 32, 32)
button.TextColor3 = Color3.fromRGB(255, 255, 255)
button.TextSize = 14
button.Font = Enum.Font.GothamBold
button.Parent = frame

local buttonCorner = Instance.new("UICorner")
buttonCorner.CornerRadius = UDim.new(0, 6)
buttonCorner.Parent = button

local function refreshButton()
	if enabled then
		button.Text = "ON"
		button.BackgroundColor3 = Color3.fromRGB(30, 90, 40)
	else
		button.Text = "OFF"
		button.BackgroundColor3 = Color3.fromRGB(90, 35, 35)
	end
end

refreshButton()

button.MouseButton1Click:Connect(function()
	enabled = not enabled
	refreshButton()
	applyState()
end)

-- Drag
local dragging = false
local dragStart
local startPos

title.InputBegan:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = true
		dragStart = input.Position
		startPos = frame.Position
	end
end)

title.InputEnded:Connect(function(input)
	if input.UserInputType == Enum.UserInputType.MouseButton1 then
		dragging = false
	end
end)

UIS.InputChanged:Connect(function(input)
	if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
		local delta = input.Position - dragStart
		frame.Position = UDim2.fromOffset(
			startPos.X.Offset + delta.X,
			startPos.Y.Offset + delta.Y
		)
	end
end)

-- Watch tools
local function hookContainer(container)
	container.ChildAdded:Connect(function(child)
		if child:IsA("Tool") then
			task.defer(function()
				if enabled then
					patchTool(child)
				end
			end)
		end
	end)

	container.ChildRemoved:Connect(function(child)
		if child:IsA("Tool") then
			restoreTool(child)
		end
	end)
end

local backpack = player:WaitForChild("Backpack")
hookContainer(backpack)

local function onCharacter(char)
	hookContainer(char)
	task.defer(applyState)
end

if player.Character then
	onCharacter(player.Character)
end
player.CharacterAdded:Connect(onCharacter)

-- Late init safety
RunService.RenderStepped:Connect(function()
	if enabled then
		for _, tool in ipairs(getAllTools()) do
			if not patchedTools[tool] then
				patchTool(tool)
			end
		end
	end
end)
